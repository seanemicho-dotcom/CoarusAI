import json
import os
from flask import Flask, render_template, request, jsonify, redirect
from dotenv import load_dotenv
from openai import OpenAI

load_dotenv()

app = Flask(__name__)

# Load tools catalog once at startup
with open("tools.json", "r", encoding="utf-8") as f:
    TOOLS = json.load(f)

# Initialize OpenAI client (set OPENAI_API_KEY in Replit Secrets)
OPENAI_API_KEY = os.getenv("OPENAI_API_KEY")
client = OpenAI(api_key=OPENAI_API_KEY)


@app.route("/")
def index():
    return render_template("index.html")


@app.route("/api/match", methods=["POST"])
def match_tools():
    """
    Receives JSON:
    {
      "userType": "smb" | "individual" | "individual_adult",
      "needs": [...],
      "budget": "...",
      "notes": "free text"
    }
    Returns: ranked tool list
    """
    data = request.json or {}
    user_type = data.get("userType")
    needs = data.get("needs", [])
    budget = data.get("budget")
    notes = data.get("notes", "")

    # Filter tools by target first (simple pre-filter)
    filtered_tools = [
        t for t in TOOLS
        if (user_type == "smb" and "smb" in t["target"]) or
           (user_type == "individual" and "individual" in t["target"]) or
           (user_type == "individual_adult" and ("individual_adult" in t["target"] or "individual" in t["target"]))
    ]

    # Build a compact representation of tools to send to the model
    tools_context = [
        {
            "id": t["id"],
            "name": t["name"],
            "description": t["description"],
            "categories": t["categories"],
            "price": t["price"]
        }
        for t in filtered_tools
    ]

    # Call LLM for ranking
    try:
        system_prompt = (
            "You are the Coarus AI Matching Engine. "
            "Your job is to rank the given tools for this user. "
            "Only recommend tools from the provided list. "
            "Return JSON ONLY in the following format:\n\n"
            "{\n"
            '  "recommendations": [\n'
            '    {\n'
            '      "id": "tool_id",\n'
            '      "score": 0-100,\n'
            '      "reason": "1-2 sentence explanation"\n'
            "    },\n"
            "    ... up to 5 tools\n"
            "  ]\n"
            "}\n\n"
            "Do NOT include any extra text outside of JSON."
        )

        user_prompt = {
            "user_type": user_type,
            "needs": needs,
            "budget": budget,
            "notes": notes,
            "tools": tools_context
        }

        # Use GPT-4.1 or a smaller model depending on your cost preference
        completion = client.chat.completions.create(
            model="gpt-4.1-mini",  # or "gpt-4.1"
            messages=[
                {"role": "system", "content": system_prompt},
                {
                    "role": "user",
                    "content": (
                        "Here is the user profile and tool list:\n\n"
                        + json.dumps(user_prompt, ensure_ascii=False, indent=2)
                    ),
                },
            ],
            temperature=0.2,
        )

        raw_output = completion.choices[0].message.content.strip()
        # Parse model JSON
        result = json.loads(raw_output)
        recs = result.get("recommendations", [])
    except Exception as e:
        print("Error calling LLM:", e)
        # Fallback: naive ordering if LLM fails
        recs = [
            {"id": t["id"], "score": 50, "reason": "Fallback recommendation."}
            for t in filtered_tools[:5]
        ]

    # Enrich with full tool data
    id_to_tool = {t["id"]: t for t in TOOLS}
    enriched = []
    for r in recs:
        tool = id_to_tool.get(r["id"])
        if not tool:
            continue
        enriched.append({
            "id": tool["id"],
            "name": tool["name"],
            "price": tool["price"],
            "reason": r.get("reason", ""),
            "score": r.get("score", 0),
            "redirectUrl": f"/go/{tool['id']}"
        })

    return jsonify({"recommendations": enriched})


@app.route("/go/<tool_id>")
def go_tool(tool_id):
    """
    Referral/redirect route.
    Later: log click to DB or analytics.
    Now: just redirect to homepage from tools.json.
    """
    tool = next((t for t in TOOLS if t["id"] == tool_id), None)
    if not tool:
        return redirect("/")

    # TODO: add logging here if needed
    url = tool.get("homepage", "/")
    return redirect(url)

document.addEventListener("DOMContentLoaded", () => {

  findBtn.addEventListener("click", async () => {

      needs,

      notes
    };

    resultsList.innerHTML = "Matching AI solutions...";
    resultsSection.style.display = "block";

    try {
      const res = await fetch("/api/match", {
        method: "POST",
        headers: {
          "Content-Type": "application/json"
        },
        body: JSON.stringify(payload)
      });

      if (!res.ok) {
        throw new Error("Server error");
      }

      const data = await res.json();
      const recs = data.recommendations || [];

      if (recs.length === 0) {
        resultsList.innerHTML = "<p>No matches found yet. Try adjusting your needs.</p>";
        return;
      }

      resultsList.innerHTML = "";
      recs.forEach(rec => {
        const div = document.createElement("div");
        div.className = "result-card";

        div.innerHTML = `
          <h3>${rec.name}</h3>
          <div class="result-meta">
            Score: ${rec.score} | Price: ${rec.price}
          </div>
          <p>${rec.reason}</p>
          <button onclick="window.location.href='${rec.redirectUrl}'">Go to Tool</button>
        `;

        resultsList.appendChild(div);
      });
    } catch (err) {
      console.error(err);
      resultsList.innerHTML = "<p>Something went wrong while matching. Please try again.</p>";
    }
  });
});

@app.route("/api/match", methods=["POST"])
def match_tools():
    """
    Receives:
    {
      "userType": "smb" | "individual" | "individual_adult",
      "needs": [...],
      "budget": "low|mid|high|enterprise",
      "notes": "string"
    }
    Returns: ranked tool list.
    """
    data = request.json or {}
    user_type = data.get("userType")
    needs = data.get("needs", [])
    budget = data.get("budget")
    notes = data.get("notes", "")

    # Filter tools by user type
    filtered_tools = [
        t for t in TOOLS
        if (user_type == "smb" and "smb" in t["target"]) or
           (user_type == "individual" and "individual" in t["target"]) or
           (user_type == "individual_adult" and ("individual_adult" in t["target"] or "individual" in t["target"]))
    ]

    if not filtered_tools:
        return jsonify({"recommendations": []})

    tools_context = [
        {
            "id": t["id"],
            "name": t["name"],
            "description": t["description"],
            "categories": t["categories"],
            "price": t["price"]
        }
        for t in filtered_tools
    ]

    system_prompt = (
        "You are the Coarus AI Matching Engine. "
        "Your job is to rank the given tools for this user. "
        "Only recommend tools from the provided list. "
        "Return JSON ONLY in this format:\n\n"
        "{\n"
        '  \"recommendations\": [\n'
        "    {\"id\": \"tool_id\", \"score\": 0-100, \"reason\": \"1-2 sentence explanation\"},\n"
        "    ... up to 5\n"
        "  ]\n"
        "}\n\n"
        "Do NOT include anything outside of valid JSON."
    )

    user_prompt = {
        "user_type": user_type,
        "needs": needs,
        "budget": budget,
        "notes": notes,
        "tools": tools_context
    }

    try:
        completion = client.chat.completions.create(
            model="gpt-4.1-mini",  # you can upgrade to gpt-4.1
            messages=[
                {"role": "system", "content": system_prompt},
                {
                    "role": "user",
                    "content": "User profile and tools:\n\n"
                               + json.dumps(user_prompt, ensure_ascii=False, indent=2)
                }
            ],
            temperature=0.2,
        )
        raw_output = completion.choices[0].message.content.strip()
        result = json.loads(raw_output)
        recs = result.get("recommendations", [])
    except Exception as e:
        print("Error calling LLM:", e)
        # Fallback: naive
        recs = [
            {"id": t["id"], "score": 50, "reason": "Fallback recommendation."}
            for t in filtered_tools[:5]
        ]

    id_to_tool = {t["id"]: t for t in TOOLS}
    enriched = []
    for r in recs:
        tool = id_to_tool.get(r["id"])
        if not tool:
            continue
        enriched.append({
            "id": tool["id"],
            "name": tool["name"],
            "price": tool["price"],
            "reason": r.get("reason", ""),
            "score": r.get("score", 0),
            "redirectUrl": f"/go/{tool['id']}"
        })

    return jsonify({"recommendations": enriched})

    if not tool:
        return redirect("/")
    url = tool.get("homepage", "/")
    print(f"[CLICK] tool_id={tool_id} -> {url}")
    return redirect(url)
